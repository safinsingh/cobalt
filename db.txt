table teams {
	id					int
	alias				string
	points				int

	unique (alias)
}

table services {
	id					int
	team_id				int (foreign->teams.id)
	alias				string

	consecutive_downs	uint
	up					boolean

	short_error       	string?
	verbose_error       string?

	unique (team_id, alias)
}

table team_snapshots {
	id					int
	team_id				int (foreign->teams.id)

    points				int
	services			jsonb ({ [service_id: int]: { service_alias: string, up: boolean } })
	time                datetime

    unique (id)
}

table sla_violations {
	id					int
	team_id				int (foreign->teams.id)
	service_id			int (foreign->service.id)
	time				datetime

    unique (id)
}
