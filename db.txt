table service_checks {
	team				string
	service				string
	up					boolean

	short_error       	string?
	long_error			string?
}

table team_snapshots {
	team				string
    points				int
	services			jsonb ({ [service_id: int]: { service_alias: string, up: boolean } })
	time                datetime

	unique (id)
}

table sla_violations {
	id					int
	team_id				int (foreign->teams.id)
	service_id			int (foreign->service.id)
	time				datetime

    unique (id)
}
