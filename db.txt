table team_snapshots {
    alias               string
    points              int
    time                datetime
}

table services {
    team                string
    service             string
    consecutive_downs   uint

    unique (team, service)
}

table uptime {
    team                string
    service             string
    up                  boolean
    error               string?

    unique (team, service)
}

table sla_violations {
    team                string
    service             string
    time                datetime

    unique (team, service)
}
