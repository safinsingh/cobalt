table teams {
	id					int
	alias				string

	unique (alias)
}

table services {
	id					int
	team_id				int (foreign->teams.id)

	alias				string
	consecutive_downs	uint

	unique (team_id, alias)
}

table team_snapshots {
	id					int
	team_id				int (foreign->teams.id)

    points				int
	time                datetime
}

table uptime {
    team                string
    service             string
    up                  boolean
    error               string?

    unique (team, service)
}

table sla_violations {
    team                string
    service             string
    time                datetime

    unique (team, service)
}
